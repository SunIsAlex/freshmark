$$
r_1(x_1-a_1)+r_2(x_2-a_2)+...+r_n(x_n-a_n)\le\sqrt{x_1^2+x_2^2+...+x_n^2}-\sqrt{a_1^2+a_2^2+...+a_n^2}
$$
